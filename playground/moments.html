Statistical Moments

modeSeries · skewSeries · kurtosisSeries · semSeries and their DataFrame equivalents — mirrors pandas.Series.mode(), .skew(), .kurt(), and .sem().

1 · Mode — most frequent value(s)

modeSeries(s) returns a new Series containing all values tied for most frequent, in sorted order. Ties are broken alphabetically / numerically. Missing values are excluded by default (dropna=true). 

import { Series, modeSeries } from "tsb";

const s = new Series({ data: [1, 2, 2, 3, 3, 4] });
const m = modeSeries(s);
console.log("modes:", m.values);   // [2, 3]
console.log("size:", m.size);      // 2

// string series
const words = new Series({ data: ["cat", "dog", "cat", "fish"] });
console.log("mode:", modeSeries(words).values); // ["cat"]

// with nulls (excluded by default)
const mixed = new Series({ data: [null, 1, 1, null, null, 2] });
console.log("dropna=true:", modeSeries(mixed).values);   // [1] — nulls skipped
console.log("dropna=false:", modeSeries(mixed, false).values); // [null] — null wins

DataFrame.mode()

import { DataFrame, Series, modeDataFrame } from "tsb";

const df = new DataFrame(new Map([
  ["score", new Series({ data: [85, 90, 85, 70, 90] })],
  ["grade", new Series({ data: ["B", "A", "B", "C", "A"] })],
]));

const modes = modeDataFrame(df);
console.log("score modes:", modes.col("score").values);  // [85, 90]
console.log("grade modes:", modes.col("grade").values);  // ["A", "B"]
console.log("shape:", modes.nrows, "×", modes.ncols);    // 2 × 2

// row-wise
const rw = modeDataFrame(df, { axis: 1 });
console.log("row modes:", rw.col("0").values);

2 · Skewness — asymmetry of the distribution

Uses the adjusted Fisher-Pearson coefficient (same formula as pandas): skew = n / ((n−1)(n−2)) × Σ((xᵢ−x̄)/s)³. Positive → right tail; negative → left tail; 0 → symmetric. Requires at least 3 non-missing values. 

import { Series, skewSeries } from "tsb";

const sym   = new Series({ data: [1, 2, 3, 4, 5] });
const right = new Series({ data: [1, 2, 3, 4, 10] });
const left  = new Series({ data: [-10, 1, 2, 3, 4] });

console.log("symmetric:", skewSeries(sym).toFixed(4));   // 0.0000
console.log("right-skewed:", skewSeries(right).toFixed(4));  // > 0
console.log("left-skewed:", skewSeries(left).toFixed(4));    // < 0

// needs ≥ 3 values
console.log("n=2:", skewSeries(new Series({ data: [1, 2] }))); // NaN

DataFrame.skew()

import { DataFrame, Series, skewDataFrame } from "tsb";

const df = new DataFrame(new Map([
  ["a", new Series({ data: [1, 2, 3, 4, 5] })],
  ["b", new Series({ data: [1, 2, 3, 4, 10] })],
]));

const sk = skewDataFrame(df);
console.log("a skew:", Number(sk.at("a")).toFixed(4));  // 0.0000
console.log("b skew:", Number(sk.at("b")).toFixed(4));  // > 0

3 · Kurtosis — tail heaviness (excess)

Returns excess kurtosis (normal distribution = 0). Uses the unbiased estimator matching pandas' Series.kurt(). Requires at least 4 non-missing values. Positive → heavy tails (leptokurtic); negative → light tails (platykurtic). 

import { Series, kurtosisSeries } from "tsb";

// uniform 1–5: platykurtic (negative excess kurtosis)
const s = new Series({ data: [1, 2, 3, 4, 5] });
console.log("kurtosis [1..5]:", kurtosisSeries(s).toFixed(4));  // -1.3000

// heavy-tailed outlier
const heavy = new Series({ data: [0, 0, 0, 0, 0, 10] });
console.log("heavy-tailed:", kurtosisSeries(heavy).toFixed(4)); // positive

// needs ≥ 4 values
console.log("n=3:", kurtosisSeries(new Series({ data: [1, 2, 3] }))); // NaN

DataFrame.kurt()

import { DataFrame, Series, kurtosisDataFrame } from "tsb";

const df = new DataFrame(new Map([
  ["uniform", new Series({ data: [1, 2, 3, 4, 5] })],
  ["peaked",  new Series({ data: [0, 0, 0, 0, 10] })],
]));

const k = kurtosisDataFrame(df);
console.log("uniform:", Number(k.at("uniform")).toFixed(4));  // -1.3000
console.log("peaked:", Number(k.at("peaked")).toFixed(4));    // positive

4 · Standard Error of the Mean (SEM)

sem = std(ddof) / √n. Default ddof=1 (sample estimate). Measures how precisely the sample mean estimates the population mean — smaller is better. 

import { Series, semSeries } from "tsb";

const s = new Series({ data: [2, 4, 4, 4, 5, 5, 7, 9] });
console.log("sem:", semSeries(s).toFixed(6));   // ≈ 0.750000

// larger sample → smaller SEM
const big = new Series({ data: Array.from({ length: 100 }, () => Math.random() * 10) });
const small = new Series({ data: Array.from({ length: 10 }, () => Math.random() * 10) });
console.log("big n:", semSeries(big).toFixed(4));
console.log("small n:", semSeries(small).toFixed(4));

// ddof=0 for population SEM
console.log("ddof=0:", semSeries(s, { ddof: 0 }).toFixed(6));

DataFrame.sem()

import { DataFrame, Series, semDataFrame } from "tsb";

const df = new DataFrame(new Map([
  ["a", new Series({ data: [1, 2, 3, 4, 5] })],
  ["b", new Series({ data: [2, 4, 6, 8, 10] })],
]));

const s = semDataFrame(df);
console.log("a sem:", Number(s.at("a")).toFixed(6));  // ≈ 0.707107
console.log("b sem:", Number(s.at("b")).toFixed(6));  // ≈ 1.414214
// b = 2*a so sem(b) = 2*sem(a)
