MultiIndex

Hierarchical multi-level index — mirrors pandas.MultiIndex.

1 — Create from tuples

The most common way: pass an array of label-tuples.

import { MultiIndex } from "tsb";

const mi = MultiIndex.fromTuples([
  ["bar", "one"],
  ["bar", "two"],
  ["baz", "one"],
  ["baz", "two"],
], { names: ["first", "second"] });

console.log(mi.nlevels);   // 2
console.log(mi.size);      // 4
console.log(mi.names);     // ["first", "second"]
console.log(mi.at(0));     // ["bar", "one"]
console.log(mi.at(-1));    // ["baz", "two"]
▶ run

2 — Create from arrays

Supply one array per level — a column-oriented alternative to fromTuples.

import { MultiIndex } from "tsb";

const mi = MultiIndex.fromArrays(
  [
    ["a", "a", "b", "b"],
    [1, 2, 1, 2],
  ],
  { names: ["letter", "number"] },
);

console.log(mi.toArray());
// [["a",1], ["a",2], ["b",1], ["b",2]]
▶ run

3 — Create from Cartesian product

fromProduct generates every combination from a list of iterables — very handy for experiment grids.

import { MultiIndex } from "tsb";

const mi = MultiIndex.fromProduct(
  [["control", "treatment"], ["pre", "post"]],
  { names: ["group", "period"] },
);

console.log(mi.size);      // 4
for (const t of mi) {
  console.log(t.join(" / "));
}
▶ run

4 — Look-up by label

Use getLoc to find the position of a tuple. contains for a quick existence check.

import { MultiIndex } from "tsb";

const mi = MultiIndex.fromTuples([
  ["a", 1], ["a", 2], ["b", 1], ["b", 2],
]);

console.log(mi.getLoc(["a", 2]));   // 1
console.log(mi.contains(["b", 3])); // false

// Duplicate tuples return an array of positions
const dup = MultiIndex.fromTuples([["a", 1], ["b", 2], ["a", 1]]);
console.log(dup.getLoc(["a", 1]));  // [0, 2]
▶ run

5 — Level operations: droplevel and swaplevel

Restructure the level hierarchy without touching the data.

import { MultiIndex } from "tsb";

const mi = MultiIndex.fromTuples([
  ["a", 1, "x"],
  ["b", 2, "y"],
], { names: ["letter", "num", "code"] });

// Drop the first level — returns a 2-level MultiIndex
const two = mi.droplevel(0) as MultiIndex;
console.log(two.nlevels);    // 2
console.log(two.at(0));      // [1, "x"]

// Swap levels 0 and 1
const swapped = mi.swaplevel(0, 1);
console.log(swapped.at(0));  // [1, "a", "x"]

// Drop to a single level → plain Index
const idx = mi.droplevel([0, 1]);
console.log(idx.constructor.name); // Index
▶ run

6 — Set operations

union, intersection, difference — same semantics as pandas.MultiIndex.

import { MultiIndex } from "tsb";

const mi1 = MultiIndex.fromTuples([["a", 1], ["b", 2], ["c", 3]]);
const mi2 = MultiIndex.fromTuples([["b", 2], ["d", 4]]);

console.log(mi1.union(mi2).toArray());
// [["a",1],["b",2],["c",3],["d",4]]

console.log(mi1.intersection(mi2).toArray());
// [["b",2]]

console.log(mi1.difference(mi2).toArray());
// [["a",1],["c",3]]
▶ run

7 — Sorting and deduplication

import { MultiIndex } from "tsb";

const mi = MultiIndex.fromTuples([
  ["b", 2], ["a", 1], ["b", 1], ["a", 2], ["a", 1],
]);

console.log(mi.sortValues().toArray());
// [["a",1],["a",1],["a",2],["b",1],["b",2]]

console.log(mi.dropDuplicates().toArray());
// [["b",2],["a",1],["b",1],["a",2]]

console.log(mi.duplicated("first"));
// [false, false, false, false, true]

8 — Missing values

import { MultiIndex } from "tsb";

const mi = MultiIndex.fromTuples([
  [null, 1],
  ["a",  2],
  ["b",  null],
  ["c",  3],
]);

console.log(mi.isna());   // [true, false, true, false]
const clean = mi.dropna();
console.log(clean.size);  // 2
console.log(clean.toArray()); // [["a",2], ["c",3]]
