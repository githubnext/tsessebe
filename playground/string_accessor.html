Series.str — Vectorised String Operations

Series.str gives you element-wise string operations on a Series, mirroring pandas StringMethods. Every method propagates null / NaN unchanged.

1 — Case operations

Convert strings to lower, upper, title, or capitalized form. Also available: swapcase(). 

const s = new tsb.Series({ data: ["hello", "WORLD", "foo BAR", null] });

console.log("lower :", s.str.lower().toArray());
console.log("upper :", s.str.upper().toArray());
console.log("title :", s.str.title().toArray());
console.log("capitalize:", s.str.capitalize().toArray());
console.log("swapcase  :", s.str.swapcase().toArray());


    


    

      
2 — Length & slicing

      

        Get string length, extract substrings with slice(), or
        access individual characters with get().
      

      


const s = new tsb.Series({ data: ["hello", "world", "pandas"] });

console.log("len        :", s.str.len().toArray());
console.log("slice(1,4) :", s.str.slice(1, 4).toArray());
console.log("get(0)     :", s.str.get(0).toArray());   // first char
console.log("get(-1)    :", s.str.get(-1).toArray());  // last char
console.log("slice step :", s.str.slice(0, undefined, 2).toArray()); // every 2nd char


      

      
      

    


    

      
3 — Strip & pad

      

        Remove whitespace (or specific characters) with strip(),
        lstrip(), rstrip().  Pad strings with
        pad(), ljust(), rjust(),
        center(), or zfill().
      

      


const s = new tsb.Series({ data: ["  hello  ", "***world***", "42"] });

console.log("strip()        :", s.str.strip().toArray());
console.log("strip('*')     :", s.str.strip("*").toArray());
console.log("ljust(10)      :", s.str.ljust(10).toArray());
console.log("rjust(10)      :", s.str.rjust(10).toArray());
console.log("center(10)     :", s.str.center(10).toArray());
console.log("zfill(5)       :",
  new tsb.Series({ data: ["42", "-7", "3"] }).str.zfill(5).toArray());


      

      
      

    


    

      
4 — Search & match

      

        Test membership with contains(), startswith(),
        endswith().  Use match() for anchored-start
        matching or fullmatch() for whole-string matching.
      

      


const s = new tsb.Series({ data: ["foo123", "bar456", "foo_bar", null] });

console.log("contains('foo')   :", s.str.contains("foo").toArray());
console.log("contains('foo', false):", // literal, not regex
  s.str.contains("foo", false).toArray());
console.log("startswith('foo') :", s.str.startswith("foo").toArray());
console.log("endswith('3')     :", s.str.endswith("3").toArray());
console.log("match('\\\\d+')    :", s.str.match("\\d+").toArray());
console.log("fullmatch('\\\\w+') :", s.str.fullmatch("\\w+").toArray());


      

      
      

    


    

      
5 — Count, find & replace

      

        Count pattern occurrences with count(), find positions with
        find() / rfind(), and replace with
        replace().
      

      


const s = new tsb.Series({ data: ["banana", "apple", "abracadabra"] });

console.log("count('a')     :", s.str.count("a").toArray());
console.log("find('a')      :", s.str.find("a").toArray()); // first index
console.log("rfind('a')     :", s.str.rfind("a").toArray()); // last index
console.log("replace('a','X'):", s.str.replace("a", "X").toArray()); // all
console.log("replace n=1   :", s.str.replace("a", "X", 1).toArray()); // first only


      

      
      

    


    

      
6 — Extract

      

        Extract the first regex capture group with extract().
        Returns null when there is no match.
      

      


const s = new tsb.Series({ data: ["price: 42", "cost: 99.5", "free"] });

// Extract numeric part after ": "
const prices = s.str.extract(":\\s*([\\d.]+)");
console.log("extracted:", prices.toArray());


      

      
      

    


    

      
7 — Split & join

      

        Split strings with split() (returns JSON array or the nth
        segment), and reassemble with join().  Use
        cat() to concatenate element-wise.
      

      


const s = new tsb.Series({ data: ["a,b,c", "x,y,z"] });

// Get the nth segment (0-based)
console.log("split n=0:", s.str.split(",", 0).toArray());
console.log("split n=1:", s.str.split(",", 1).toArray());

// Full split → JSON array → rejoin
const joined = s.str.split(",").str.join(" | ");
console.log("rejoin   :", joined.toArray());

// Concatenate two series element-wise
const a = new tsb.Series({ data: ["foo", "bar"] });
const b = new tsb.Series({ data: ["1", "2"] });
console.log("cat      :", a.str.cat([b.toArray()], "-").toArray());


      

      
      

    


    

      
8 — Predicates

      

        Test character classes with isalpha(),
        isdigit(), isalnum(), islower(),
        isupper(), istitle(), isspace().
      

      


const s = new tsb.Series({ data: ["abc", "123", "abc123", "ABC", "Hello World", "   "] });

console.log("isalpha :", s.str.isalpha().toArray());
console.log("isdigit :", s.str.isdigit().toArray());
console.log("isalnum :", s.str.isalnum().toArray());
console.log("isupper :", s.str.isupper().toArray());
console.log("istitle :", s.str.istitle().toArray());
console.log("isspace :", s.str.isspace().toArray());


      

      
      

    


    

      
9 — Null propagation

      

        All methods pass null / NaN through unchanged,
        just like pandas.
      

      


const s = new tsb.Series({ data: ["hello", null, "world"] });

console.log("upper:", s.str.upper().toArray());
console.log("len  :", s.str.len().toArray());
console.log("contains:", s.str.contains("o").toArray());
