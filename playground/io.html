tsb provides CSV and JSON I/O utilities that mirror pandas.read_csv, pandas.read_json, DataFrame.to_csv, and DataFrame.to_json.

readCsv read_csv

Parse a CSV string into a DataFrame. Handles quoted fields, custom separators, missing-value markers, and integer/float/boolean type inference.

Basic CSV parsing
import { readCsv } from "tsb";

const csv = `name,age,score
Alice,30,95.5
Bob,25,88.0
Carol,35,91.2`;

const df = readCsv(csv);
df.shape;          // [3, 3]
df.col("name").values;  // ["Alice", "Bob", "Carol"]
df.col("age").values;   // [30, 25, 35]   ← inferred as integers
df.col("score").values; // [95.5, 88.0, 91.2] ← inferred as floats
Missing values & custom separator
const tsv = `a\tb\tc
1\t\t3
\t5\t6`;

const df = readCsv(tsv, { sep: "\t" });
df.col("b").values;   // [null, 5] — empty cell → null

// Custom NA values
const df2 = readCsv("a,b\nMISSING,1\n2,3", { naValues: ["MISSING"] });
df2.col("a").values;  // [null, 2]
Index column & row limit
const csv = `id,name,value
r1,Alice,10
r2,Bob,20
r3,Carol,30`;

const df = readCsv(csv, {
  indexCol: "id",   // use "id" column as row index
  nrows: 2,         // only read 2 data rows
});
df.index.values;   // ["r1", "r2"]
df.columns.values; // ["name", "value"]
df.shape;          // [2, 2]
Quoted fields
const csv = `name,bio
Alice,"She said ""hello, world!"""
Bob,"Multi-word, description"`;

const df = readCsv(csv);
df.col("bio").at(0);
// 'She said "hello, world!"'

toCsv to_csv

Serialize a DataFrame (or Series) to a CSV string.

Basic serialization
import { DataFrame, toCsv } from "tsb";

const df = DataFrame.fromColumns({
  name:  ["Alice", "Bob"],
  score: [95, 88],
});

toCsv(df);
// ",name,score\n0,Alice,95\n1,Bob,88\n"

toCsv(df, { index: false });
// "name,score\nAlice,95\nBob,88\n"
Options: naRep, columns, sep
const df = DataFrame.fromColumns({ a: [1, null, 3], b: [4, 5, 6] });

toCsv(df, { naRep: "NaN", index: false });
// "a,b\n1,4\nNaN,5\n3,6\n"

toCsv(df, { sep: ";", columns: ["a"], index: false });
// "a\n1\nNaN\n3\n"   (only column "a", semicolon sep)

readJson read_json

Parse JSON into a DataFrame. Supports five orientations matching pandas.

orient="records" (default)
import { readJson } from "tsb";

const json = `[
  {"name": "Alice", "score": 95},
  {"name": "Bob",   "score": 88}
]`;

const df = readJson(json);
df.shape;                   // [2, 2]
df.col("name").values;  // ["Alice", "Bob"]
orient="split"
const json = JSON.stringify({
  index:   ["r1", "r2"],
  columns: ["a", "b"],
  data:    [[1, 2], [3, 4]],
});

const df = readJson(json, { orient: "split" });
df.index.values;    // ["r1", "r2"]
df.col("a").values; // [1, 3]
All orientations
// orient="records"  → [{col: val, ...}, ...]
// orient="columns"  → {col: {idx: val, ...}, ...}
// orient="values"   → [[v, ...], ...]  (2-D array)
// orient="index"    → {idx: {col: val, ...}, ...}
// orient="split"    → {index, columns, data}

// values orientation needs column names provided:
readJson("[[1,2],[3,4]]", {
  orient: "values",
  columns: ["x", "y"],
});

toJson to_json

Serialize a DataFrame to JSON. Supports six orientations.

Basic serialization
import { DataFrame, toJson } from "tsb";

const df = DataFrame.fromColumns({ a: [1, 2], b: [3, 4] });

toJson(df, { orient: "records" });
// '[{"a":1,"b":3},{"a":2,"b":4}]'

toJson(df, { orient: "split" });
// '{"index":[0,1],"columns":["a","b"],"data":[[1,3],[2,4]]}'

toJson(df, { orient: "values" });
// '[[1,3],[2,4]]'
Round-trip: toCsv → readCsv
const df = DataFrame.fromColumns({
  x: [1, 2, 3],
  y: [4.5, null, 6.7],
  z: ["a", "b", "c"],
});

const csv  = toCsv(df, { index: false });
const df2  = readCsv(csv);
df2.col("x").values; // [1, 2, 3]  ← integers preserved
df2.col("y").values; // [4.5, null, 6.7]  ← nulls preserved
df2.col("z").values; // ["a", "b", "c"]

// JSON round-trip
const json = toJson(df, { orient: "records" });
const df3  = readJson(json);
df3.shape; // [3, 3]
