pivot & pivotTable

Reshape DataFrames — mirrors pandas.DataFrame.pivot() and pandas.pivot_table().

1 — pivot: basic reshape

pivot reshapes a DataFrame using unique values in one column as the new column headers. Requires one unique value per (row, column) pair.

import { DataFrame, pivot } from "tsb";

const df = DataFrame.fromColumns({
  date:  ["2020-01", "2020-01", "2020-02", "2020-02"],
  city:  ["NYC",     "LA",      "NYC",     "LA"],
  temp:  [50,        60,        55,        65],
});

const wide = pivot(df, { index: "date", columns: "city", values: "temp" });
console.log(wide.toRecords());
// [{ NYC: 50, LA: 60 }, { NYC: 55, LA: 65 }]
console.log([...wide.index.values]);
// ["2020-01", "2020-02"]

2 — pivot: multiple value columns

When values is omitted, all non-index/non-column columns are used. Column names become valCol_colHdr.

import { DataFrame, pivot } from "tsb";

const df = DataFrame.fromColumns({
  row:  ["A", "A", "B", "B"],
  col:  ["x", "y", "x", "y"],
  val1: [1,   2,   3,   4],
  val2: [10,  20,  30,  40],
});

const wide = pivot(df, { index: "row", columns: "col" });
console.log(wide.columns.values);
// ["val1_x", "val1_y", "val2_x", "val2_y"]
console.log(wide.toRecords());
▶ run

3 — pivotTable: mean aggregation

pivotTable aggregates values when multiple rows map to the same (index, column) cell.

import { DataFrame, pivotTable } from "tsb";

const df = DataFrame.fromColumns({
  region:  ["East", "East", "East", "West", "West", "West"],
  product: ["A",    "A",    "B",    "A",    "B",    "B"],
  sales:   [100,    120,    80,     150,    90,     110],
});

const table = pivotTable(df, {
  index: "region",
  columns: "product",
  values: "sales",
  aggfunc: "mean",
});
console.log(table.toRecords());
// East: A=110, B=80; West: A=150, B=100
▶ run

4 — pivotTable: sum with fill_value

Use aggfunc: "sum" to total up values per cell, and fill_value to replace missing cells.

import { DataFrame, pivotTable } from "tsb";

const df = DataFrame.fromColumns({
  dept:    ["eng", "eng", "mktg", "mktg"],
  quarter: ["Q1",  "Q2",  "Q1",   "Q2"],
  budget:  [500,   600,   300,    350],
});

const table = pivotTable(df, {
  index: "dept",
  columns: "quarter",
  values: "budget",
  aggfunc: "sum",
  fill_value: 0,
});
console.log(table.toRecords());
▶ run

5 — pivotTable: count

Use aggfunc: "count" to count how many rows fall into each cell.

import { DataFrame, pivotTable } from "tsb";

const df = DataFrame.fromColumns({
  gender:    ["M", "F", "M", "F", "M", "F"],
  category:  ["A", "A", "B", "B", "A", "B"],
  value:     [1,   2,   3,   4,   5,   6],
});

const counts = pivotTable(df, {
  index: "gender",
  columns: "category",
  values: "value",
  aggfunc: "count",
});
console.log(counts.toRecords());
// M: A=2, B=1; F: A=1, B=2
